# --- Idle Shutdown ---
# Sidecar container monitors active connections and scales service to 0 when idle.
ENABLE_IDLE_SHUTDOWN=true
IDLE_TIMEOUT_MINUTES=30
IDLE_CHECK_INTERVAL_SECONDS=60
# Detection method: "netstat" checks network connections on CONTAINER_PORT,
# "http" queries a status endpoint for an active player/connection count.
IDLE_CHECK_METHOD=netstat
# Only used when IDLE_CHECK_METHOD=http. Must return JSON with a numeric
# "connections" or "players" field. Example: { "players": 3 }
IDLE_STATUS_ENDPOINT=http://localhost:7777/status

# --- Secrets ---
# Secrets are injected as ECS secrets — never plaintext env vars, task-def fields, or logs.
# Each ref maps a container env var to a value stored in AWS Secrets Manager (sm)
# or SSM Parameter Store SecureString (ssm). Grammar:
#
#   SECRET_REFS=<ENV_VAR>=<sm|ssm>:<sourceId>[|<jsonKey>], ...
#
#   <ENV_VAR>    name the container receives (same effect as a GAME_ENV_ var)
#   sm:<id>      Secrets Manager secret name or ARN; append |<jsonKey> for a JSON secret field
#   ssm:<id>     SSM SecureString parameter path
#
# Note: the jsonKey delimiter is '|', not '#' — dotenv strips '#...' as an inline comment.
#
# Naming convention for sourceId:  respawn/<service>/<name>  (sm)
#                                 /respawn/<service>/<name>  (ssm)
#
# Set/rotate values out-of-band (never commit them) via the Secrets CLI action
# or the secrets-runner util — values are entered with masked input.
#
# Example: SECRET_REFS=RCON_PASSWORD=sm:respawn/example/rcon,SRV_TOKEN=ssm:/respawn/example/token
SECRET_REFS=
